```go
// @lc code=start
func divideString(s string, k int, fill byte) []string {
	res, path, n := []string{}, "", len(s)

	for i := 0; i < n; i++ {
		path += string(s[i])
		if (i+1)%k == 0 {
			res = append(res, path)
			path = path[:0]
		}
	}
	if n%k != 0 {
		for j := n % k; j < k; j++ {
			path += string(fill)
		}
		res = append(res, path)
	}
	return res
}
```

```go
func divideString(s string, k int, fill byte) []string {
	to_ret := []string{}
	for len(s) >= k {
		to_ret = append(to_ret, s[:k])
		s = s[k:]
	}
	if len(s) > 0 {
		to_ret = append(to_ret, s[:])
		s = s[:0]
	}
	to_ret[len(to_ret)-1] += strings.Repeat(string(fill), k-len(to_ret[len(to_ret)-1]))
	return to_ret
}
```

[5194. 得到目标值的最少行动次数](https://leetcode-cn.com/contest/weekly-contest-276/problems/minimum-moves-to-reach-target-score/)

```go
func minMoves(target int, maxDoubles int) int {
    to_ret := 0
    for target > 1 {
        if maxDoubles == 0 {
            to_ret += target - 1
            break
        }
        if target % 2 == 1 {
            to_ret ++
            target --
        } else {
            to_ret ++
            target = target / 2
            maxDoubles --
        }
    }
    return to_ret
}
```

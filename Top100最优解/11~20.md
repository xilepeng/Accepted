## [5. 最长回文子串](https://leetcode-cn.com/problems/longest-palindromic-substring/)


```go
func longestPalindrome(s string) string {
	res, n := "", len(s)
	for i := 0; i < n; i++ {
		// n 为偶数
		l, r := i, i
		for l >= 0 && r < n && s[l] == s[r] {
			if len(res) < r-l+1 {
				res = s[l : r+1]
			}
			l--
			r++
		}
		// n 为奇数
		l, r = i, i+1
		for l >= 0 && r < n && s[l] == s[r] {
			if len(res) < r-l+1 {
				res = s[l : r+1]
			}
			l--
			r++
		}
	}
	return res
}
```

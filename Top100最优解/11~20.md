## [5. 最长回文子串](https://leetcode-cn.com/problems/longest-palindromic-substring/)

```go
func longestPalindrome(s string) string {
	res, n := "", len(s)
	var extend func(int, int) string

	extend = func(i, j int) string { // 中心扩展算法
		sub := ""
		for i >= 0 && j < n && s[i] == s[j] {
			sub = s[i : j+1]
			i-- // 扩展
			j++
		}
		if len(res) < len(sub) {
			return sub
		}
		return res
	}

	for i := 0; i < n; i++ {
		res = extend(i, i)   // 以自身为中心点
		res = extend(i, i+1) // 以自身和自身的下一个元素为中心点
	}
	return res
}
```


```go
func longestPalindrome(s string) string {
	res, n := "", len(s)
	for i := 0; i < n; i++ {
		// n 为偶数
		l, r := i, i
		for l >= 0 && r < n && s[l] == s[r] {
			if len(res) < r-l+1 {
				res = s[l : r+1]
			}
			l--
			r++
		}
		// n 为奇数
		l, r = i, i+1
		for l >= 0 && r < n && s[l] == s[r] {
			if len(res) < r-l+1 {
				res = s[l : r+1]
			}
			l--
			r++
		}
	}
	return res
}
```
